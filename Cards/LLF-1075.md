WITH matched AS (
  SELECT
    le.FranchiseId,
    le.LoanConsultantId                                       AS le_LoanConsultantId,
    mf.LeadAllocation.Preference.LoanConsultantId             AS mf_LoanConsultantId,
    le.CustomerManagerId                                      AS le_CustomerManagerId,
    mf.LeadAllocation.Preference.CustomerManagerId            AS mf_CustomerManagerId,
    le.LeadRecipientEmail                                     AS le_LeadRecipientEmail,
    mf.LeadAllocation.Preference.NotificationEmail            AS mf_NotificationEmail,
    le.LeadRecipientSMS                                       AS le_LeadRecipientSMS,
    mf.LeadAllocation.Preference.NotificationSMS              AS mf_NotificationSMS,
    le.LeadRecipientTelephone                                 AS le_LeadRecipientTelephone,
    mf.LeadAllocation.Preference.NotificationPhone            AS mf_NotificationPhone,
    le.FranchiseAvailability                                  AS le_FranchiseAvailability,
    mf.LeadAllocation.Preference.IsAvailableforLeadAllocation AS mf_IsAvailableForLeadAllocation,
    le.AlternateFranchiseId                                   AS le_AlternateFranchiseId,
    mf.LeadAllocation.Preference.AlternateFranchiseId         AS mf_AlternateFranchiseId,
    le.PreferredContactMethodId                               AS le_PreferredContactMethodId,
    mf.LeadAllocation.Preference.NotificationPreferenceType   AS mf_NotificationPreferenceType,
    le.FranchiseState                                         AS le_FranchiseState,
    mf.LeadAllocation.Preference.State                        AS mf_State,
    le.AutomaticQualify                                       AS le_AutomaticQualify,
    mf.LeadAllocation.Preference.AutoQualifyLead              AS mf_AutoQualifyLead,
    le.FileCreation                                           AS le_FileCreation,
    mf.LeadAllocation.Preference.AutoCreateFile               AS mf_AutoCreateFile,
    le.SendFactFind                                           AS le_SendFactFind,
    mf.LeadAllocation.Preference.AutoSendFactFind             AS mf_AutoSendFactFind,
    ARRAY_TO_STRING(ARRAY(SELECT CAST(p AS STRING) FROM UNNEST(le.Postcodes) p ORDER BY p), ',') AS le_Postcodes,
    ARRAY_TO_STRING(ARRAY(SELECT p FROM UNNEST(mf.LeadAllocation.AreaPostcode) p ORDER BY p), ',')  AS mf_AreaPostcode
  FROM `mc-data-lake-preprod-1.COMPOSITES.LE_FRANCHISES` le
  JOIN `mc-data-lake-preprod-1.STAGING_TABLES_DISCO.Master_Franchise` mf
    ON le.FranchiseId = mf.Id
)
SELECT
  FranchiseId,
  field_name,
  le_value,
  mf_value
FROM matched
CROSS JOIN UNNEST([
  STRUCT('LoanConsultantId'        AS field_name, CAST(le_LoanConsultantId        AS STRING) AS le_value, CAST(mf_LoanConsultantId             AS STRING) AS mf_value),
  STRUCT('CustomerManagerId'       AS field_name, CAST(le_CustomerManagerId       AS STRING) AS le_value, CAST(mf_CustomerManagerId            AS STRING) AS mf_value),
  STRUCT('LeadRecipientEmail'      AS field_name, le_LeadRecipientEmail                     AS le_value, mf_NotificationEmail                            AS mf_value),
  STRUCT('LeadRecipientSMS'        AS field_name, le_LeadRecipientSMS                       AS le_value, mf_NotificationSMS                              AS mf_value),
  STRUCT('LeadRecipientTelephone'  AS field_name, le_LeadRecipientTelephone                 AS le_value, mf_NotificationPhone                            AS mf_value),
  STRUCT('FranchiseAvailability'   AS field_name, CAST(le_FranchiseAvailability   AS STRING) AS le_value, CAST(mf_IsAvailableForLeadAllocation AS STRING) AS mf_value),
  STRUCT('AlternateFranchiseId'    AS field_name, CAST(le_AlternateFranchiseId    AS STRING) AS le_value, CAST(mf_AlternateFranchiseId         AS STRING) AS mf_value),
  STRUCT('PreferredContactMethodId'AS field_name, CAST(le_PreferredContactMethodId AS STRING) AS le_value, CAST(mf_NotificationPreferenceType  AS STRING) AS mf_value),
  STRUCT('FranchiseState'          AS field_name, le_FranchiseState                         AS le_value, mf_State                                        AS mf_value),
  STRUCT('AutomaticQualify'        AS field_name, CAST(le_AutomaticQualify        AS STRING) AS le_value, CAST(mf_AutoQualifyLead              AS STRING) AS mf_value),
  STRUCT('FileCreation'            AS field_name, CAST(le_FileCreation            AS STRING) AS le_value, CAST(mf_AutoCreateFile               AS STRING) AS mf_value),
  STRUCT('SendFactFind'            AS field_name, CAST(le_SendFactFind            AS STRING) AS le_value, CAST(mf_AutoSendFactFind             AS STRING) AS mf_value),
  STRUCT('Postcodes'               AS field_name, le_Postcodes                              AS le_value, mf_AreaPostcode                                 AS mf_value)
])
WHERE le_value IS DISTINCT FROM mf_value
ORDER BY FranchiseId, field_name;

WITH matched AS (

    le.LeadRecipientEmail                                     AS le_LeadRecipientEmail,

    le.LeadRecipientSMS                                       AS le_LeadRecipientSMS,

    mf.LeadAllocation.Preference.NotificationPhone            AS mf_NotificationPhone,

    le.FranchiseAvailability                                  AS le_FranchiseAvailability,
    mf.LeadAllocation.Preference.IsAvailableforLeadAllocation AS mf_IsAvailableForLeadAllocation,

    CAST(le.AlternateFranchiseId AS STRING)                   AS le_AlternateFranchiseId,
    mf.LeadAllocation.Preference.AlternateFranchiseId         AS mf_AlternateFranchiseId,

    le.PreferredContactMethodId                               AS le_PreferredContactMethodId,
    mf.LeadAllocation.Preference.NotificationPreferenceType   AS mf_NotificationPreferenceType,

    le.FranchiseState                                         AS le_FranchiseState,
    mf.LeadAllocation.Preference.State                        AS mf_State,

    le.AutomaticQualify                                       AS le_AutomaticQualify,
    mf.LeadAllocation.Preference.AutoQualifyLead              AS mf_AutoQualifyLead,

    le.FileCreation                                           AS le_FileCreation,
    mf.LeadAllocation.Preference.AutoCreateFile               AS mf_AutoCreateFile,

    le.SendFactFind                                           AS le_SendFactFind,
    mf.LeadAllocation.Preference.AutoSendFactFind             AS mf_AutoSendFactFind,

    ARRAY_TO_STRING(
      ARRAY(SELECT CAST(p AS STRING) FROM UNNEST(le.Postcodes) p ORDER BY p),
      ','
    ) AS le_Postcodes,

    ARRAY_TO_STRING(
      ARRAY(SELECT p FROM UNNEST(mf.LeadAllocation.AreaPostcode) p ORDER BY SAFE_CAST(p AS INT64)),
      ','
    ) AS mf_AreaPostcode

  FROM `mc-data-lake-preprod-1.COMPOSITES.LE_FRANCHISES` le
  JOIN `mc-data-lake-preprod-1.STAGING_TABLES_DISCO.Master_Franchise` mf
    ON le.FranchiseId = mf.Id
)

SELECT *
FROM matched
WHERE
  le_LoanConsultantId        IS DISTINCT FROM mf_LoanConsultantId
  OR le_CustomerManagerId   IS DISTINCT FROM mf_CustomerManagerId
  OR le_LeadRecipientEmail  IS DISTINCT FROM mf_NotificationEmail
  OR le_LeadRecipientSMS    IS DISTINCT FROM mf_NotificationSMS
  OR le_LeadRecipientTelephone IS DISTINCT FROM mf_NotificationPhone
  OR le_FranchiseAvailability IS DISTINCT FROM mf_IsAvailableForLeadAllocation
  OR le_AlternateFranchiseId IS DISTINCT FROM mf_AlternateFranchiseId
  OR le_PreferredContactMethodId IS DISTINCT FROM mf_NotificationPreferenceType
  OR le_FranchiseState      IS DISTINCT FROM mf_State
  OR le_AutomaticQualify    IS DISTINCT FROM mf_AutoQualifyLead
  OR le_FileCreation        IS DISTINCT FROM mf_AutoCreateFile
  OR le_SendFactFind        IS DISTINCT FROM mf_AutoSendFactFind
  OR le_Postcodes           IS DISTINCT FROM mf_AreaPostcode
ORDER BY FranchiseId;
